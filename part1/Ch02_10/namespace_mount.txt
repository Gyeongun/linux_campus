# unshare 명령어 확인
unshare --help

# 새로운 mount namespace 생성
unshare -m /bin/bash

# tmpfs 마운트
mount -t tmpfs tmpfs /mnt
mount | grep /mnt

# 테스트 파일 생성
echo "Hello" > /mnt/hello.txt
cat /mnt/hello.txt

# Terminal을 새로 열어서 테스트 파일 내용 확인
cat /mnt/hello.txt
nsenter -t $TARGET_PID -m ls /mnt

# mount namespace 종료
exit
