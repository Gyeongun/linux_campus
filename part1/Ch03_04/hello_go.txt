# 프로젝트 디렉토리 생성
mkdir -p ~/projects/hello

# 프로젝트 초기화. Go 모듈 생성
cd ~/projects/hello
go mod init example/hello

# hello.go 파일 작성
"""
package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}
"""

# hello 프로그램 실행
go run .

# 외부 패키지를 포함한 코드 실행
"""
package main

import "fmt"

import "rsc.io/quote"

func main() {
    fmt.Println(quote.Go())
}
"""

# 패키지 의존성 관리
go mod tidy

# hello 프로그램 실행
go run .
